-- ServerHopper (Teleport)
local servers = game:GetService("HttpService"):JSONDecode(
    game:HttpGet("https://games.roblox.com/v1/games/"..placeId.."/servers/Public?limit=100")
).data

local currentJobId = game.JobId
local targetServer = nil

for _, server in ipairs(servers) do
    if server.id ~= currentJobId and server.playing < 10 then
        targetServer = server.id
        break
    end
end

if targetServer then
    TeleportService:TeleportToPlaceInstance(placeId, targetServer)
else
    TeleportService:Teleport(placeId)
end
