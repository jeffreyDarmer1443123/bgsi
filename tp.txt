-- ServerHopper (Teleport-Logik)
local function fetchServers()
    local response = game:HttpGetAsync(
        "https://games.roblox.com/v1/games/"..placeId..
        "/servers/Public?sortOrder=Asc&limit=100"
    )
    return game:GetService("HttpService"):JSONDecode(response).data
end

local function findValidServer(servers)
    local currentJobId = game.JobId
    for _, server in ipairs(servers) do
        if server.id ~= currentJobId and server.playing < 10 then
            return server.id
        end
    end
end

local function hopServer()
    for i = 1, 5 do  -- Max 5 Versuche
        local servers = fetchServers()
        local target = findValidServer(servers)
        if target then
            TeleportService:TeleportToPlaceInstance(placeId, target)
            task.wait(5)  -- Wartezeit fÃ¼r Teleport
            return
        end
        task.wait(2 * i)  -- Backoff-Zeit
    end
    TeleportService:Teleport(placeId)  -- Fallback
end

print("Starte ServerHop...")
hopServer()
